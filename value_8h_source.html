00276         // Helper for path functions which provides an iterator interface for
00277         // splitting a path into parts and traversing them
00278         struct PathIterator {
00279             // The path passed in must remain valid for the entire use of the
00280             // PathIterator, it isn't copied
00281             PathIterator(const String& path_, const typename String::value_type delim_)
00282              : path(path_), delim(delim_), prev_delim_pos(-1), delim_pos(0)
00283             {
00284                 // prev_delim_pos = -1 so when we search at prev_delim_pos+1 for
00285                 // the next delimiter, it'll do the right thing by search from
00286                 // the start of the string. Note that we need to be careful with
00287                 // this since this could mean prev_delim_pos == npos.
00288 
00289                 // delim_pos just needs to be non-npos here so finished()
00290                 // returns false, even if this is an empty string. The value
00291                 // doesn't matter since as soon as we try to look at the next
00292                 // value we'll just overwrite it.
00293             }
00294 
00295             // Get the next part of the path. The return value is the subpath
00296             // and, if we've reached the end of the path, subsequent calls to
00297             // finished() will return true.
00298             String next() {
00299                 assert(!finished());
00300 
00301                 delim_pos = path.find(delim, prev_delim_pos+1);
00302                 // We should never have delim_pos == npos because of the outer
00303                 // check for stop_pos
00304                 String subpath;
00305                 if (delim_pos == String::npos)
00306                     subpath = path.substr(prev_delim_pos+1);
00307                 else
00308                     subpath = path.substr(prev_delim_pos+1, delim_pos-prev_delim_pos-1);
00309 
00310                 if (subpath.empty())
00311                     throw PathError(path, subpath, "Found empty subpath");
00312 
00313                 prev_delim_pos = delim_pos;
00314                 return subpath;
00315             }
00316 
00317             bool finished() const {
00318                 return (prev_delim_pos == String::npos && delim_pos == String::npos);
00319             }
00320 
00321             const String& path;
00322             const typename String::value_type delim;
00323             std::size_t prev_delim_pos;
00324             std::size_t delim_pos;
00325         };

00964     bool BasicValue< Config >::insert_put_impl(const typename Config::String_type& path, const BasicValue& val, const typename Config::String_type::value_type delim, bool force_val) {
00965         if (!isObject())
00966             throw PathError(path, "<root>", "Value isn't an object.");
00967 
00968         Object* obj = &(getObject());
00969 
00970         PathIterator path_it(path, delim);
00971         do {
00972             String subpath = path_it.next();
00973             if (!path_it.finished()) {
00974                 // Not the last component, get the next subobject
00975 
00976                 // Insert a new object. If an object was already
00977                 // there, we just get back the existing object to work
00978                 // with.
00979 
00980                 typename Object::iterator sub_obj_it = (obj->insert( std::make_pair(subpath, BasicValue(Object())) )).first;
00981                 if (!sub_obj_it->second.isObject())
00982                     throw PathError(path, subpath, "Path component isn't an object.");
00983                 obj = &(sub_obj_it->second.getObject());
00984             }
00985             else {
00986                 // Finally, insert the value
00987                 if (force_val)
00988                     obj->erase(subpath);
00989                 return obj->insert( std::make_pair(subpath, val) ).second;
00990             }
00991         } while(!path_it.finished());
00992 
00993         // Shouldn't get here, just have the return to make the compiler happy.
00994         return false;
00995     }
